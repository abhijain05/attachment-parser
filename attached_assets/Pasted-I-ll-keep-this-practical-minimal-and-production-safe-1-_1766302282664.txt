5️⃣ Streaming Example (Recommended for UI)
const response = await fetch("http://31.97.210.209:8001/chat", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "api-key": process.env.VPS_API_KEY
  },
  body: JSON.stringify({
    model: "medium",
    session_id: "user_123",
    prompt: "Explain this resume in detail",
    stream: true
  })
});

const reader = response.body.getReader();
const decoder = new TextDecoder();

while (true) {
  const { value, done } = await reader.read();
  if (done) break;
  console.log(decoder.decode(value));
}


✔ Perfect for chat UX
✔ No timeout issues
